events { }

http {
    upstream backend {
        server be:8081;  # Backend service
    }

    upstream web {
        server web_luan_food:5173;  # Web front-end service
    }

    upstream admin {
        server admin_luan_food:5174;  # Admin front-end service
    }

    server {
        listen 80;

        location /api/ {
            proxy_pass http://backend/;  # Proxy API requests to the backend
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        location / {
            proxy_pass http://web;  # Proxy to web front-end
        }

        location /admin/ {
            proxy_pass http://admin;  # Proxy to admin front-end
        }
    }
}
